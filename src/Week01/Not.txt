/*
Programlama: Bir programı oluşturmak için gerekli komutların

Syntax(Söz dizilimi): Programlama dilinin dil bilgisi (gramer) kurallarıdır.
                      (Örneğin java'da her bir komut satırının sonuna noktalı virgül(;) konulur)

Kaynak Kod(Source Code): Bir programlama diliyle yazılmış metinlere denir.

Makine Kodu: Bilgisayarın direkt anlayabildiği yegane dil.

Derleyici(Compiler) ve Yorumlayıcı(Interpreter): Kaynak kodu makine koduna dönüştüren araçlardır.

IDE(Integrated Development Environment): Sadece yazmayı ve düzenlemeyi sağlamakla sınırlı kalmayıp

                                         belirli bir programlama dili için özelleştirilmiş gelişmiş
                                         kod düzenleyicileridir.(intellij idea,PyCharm,...)
*/

//Veri Tipleri(Data Types):
/*
Tamsayılar:
byte (8 bit) (-128 ile 127 arasında)
short (16 bit) (-32.768 ile 32.767)
int (32 bit) (+-2.147.1483.647)
long (64 bit)

NOT: long bir değerin sonuna L yazılır(int sınırının dışındaysa zorunlu)
*/
byte sayi1=25;
short sayi2=31500;
int sayi3=56646546;
long sayi4=1231231231231L;

/*
Ondalıklı sayılar:
float (32 bit): virgülden sonra 6-7 digit
double (64 bit): virgülden sonra 15-16 digit

NOT: float bir değerin sonuna f yazılır(zorunludur)
     double bir değerin sonuna d yazılır(zorunlu değil)
*/
float sayi5=0.12312f;
double sayi6=0.12312312d;

// boolean (1 bit: 0-false veya 1-true)
boolean dogruMu=true;

/*
char (16 bit) (0 ile 65535) ASCII(8 bit)/UNICODE(8 veya 16 bit)

NOT:karakter değerleri tek tırnak içerisinde gösterilir(zorunludur)
*/
char harfNotu='A';
char buyukA=65;

/* String (non-primitive)

NOT:metinsel değerler çift tırnak içerisinde gösterilir(zorunludur)
NOT:non-primitive tiplere başka örnekler:Array,Class,Enum,Interface,...
*/
String ad = "Tayyip";


/* Type Casting (Tip Dönüşümü):
Widening Casting (Genişletme): byte -> short -> char -> int -> long -> float -> double (otomatik)
Narrowing Casting (Daraltma): double -> float -> long -> int -> short -> byte (manuel)
*/
byte sayiByte=10;
double sayiDouble=sayiByte;
int sayiInt=(int) sayiDouble;
short ascii=(short) 'a';
